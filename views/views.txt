CREATE VIEW CLIENTE_DIR
 AS 
 SELECT NOM , DIR FROM CLIENTE GROUP BY NOM , DIR

 CREATE VIEW CLENTE_A AS(SELECT NOM FROM CLIENTE GROUP BY NOM HAVING SUBSTRING(NOM,1,1)='A')
 GO
SELECT * FROM CLENTE_A
 GO
 CREATE VIEW PROD_COMPRAS AS (
SELECT PRODUCTO.COD_P AS NOMP,PRECIO,COMPRA.COD_P FROM PRODUCTO LEFT JOIN COMPRA ON CURP = COMPRA.CURP
)
GO


SELECT * FROM PROD_COMPRAS

GO
CREATE VIEW personas_compras AS(
SELECT NOM, APES,DIR,TEL FROM CLIENTE LEFT JOIN COMPRA ON CLIENTE.CURP = COMPRA.CURP
WHERE CLIENTE.CURP = (SELECT CURP FROM CLIENTE where COMPRA.CURP= CLIENTE.CURP) 
GROUP BY NOM,APES,DIR,TEL
)
GO
SELECT * FROM personas_compras
GO
CREATE VIEW PROD_PROV AS (
SELECT PROVEEDOR.NOM_PRO AS NOM_PRO,DIR_PRO
FROM PRODUCTO RIGHT JOIN PROVEEDOR ON PROVEEDOR.RFC_PRO=PRODUCTO.RFC_PRO
GROUP BY PRODUCTO.NOMP,PROVEEDOR.NOM_PRO,DIR_PRO
)
GO
SELECT * FROM PROD_PROV
